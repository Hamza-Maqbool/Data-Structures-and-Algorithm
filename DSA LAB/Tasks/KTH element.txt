class Solution {
public:
    TreeNode* searchBST(TreeNode* root, int val) {
        TreeNode* temp = root;
        if(root == NULL){
            return NULL;
        }
        if(temp->val == val)
             return temp;
        else if(val < temp->val){
            temp = searchBST(temp->left,val);
        }
        else if(val > temp->val)
            temp = searchBST(temp->right,val);
        // searchBST(temp->left,val);
         return temp;
    }
